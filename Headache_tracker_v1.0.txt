    <script>
        let currentEntry = { type: '', sev: '', med: '' };

        function select(category, value) {
            currentEntry[category] = value;
            const containerId = category === 'type' ? 'typeGroup' : (category === 'sev' ? 'sevGroup' : 'medGroup');
            const buttons = document.querySelectorAll(`#${containerId} button`);
            buttons.forEach(b => {
                b.classList.toggle('active', b.innerText === value);
            });
        }

        function saveLog() {
            if(!currentEntry.type || !currentEntry.sev) return alert("Please select Type and Severity");
            const logs = JSON.parse(localStorage.getItem('headacheLogs') || '[]');
            logs.unshift({ ...currentEntry, date: new Date().toLocaleString() });
            localStorage.setItem('headacheLogs', JSON.stringify(logs.slice(0, 10))); // Keep last 10
            renderLogs();
            alert("Saved!");
        }

        function renderLogs() {
            const logs = JSON.parse(localStorage.getItem('headacheLogs') || '[]');
            const html = logs.map(l => `<div><strong>${l.date}:</strong> ${l.type} (S:${l.sev}) - ${l.med}</div>`).join('<hr>');
            document.getElementById('historyEntries').innerHTML = html || 'No entries yet.';
        }

        window.onload = renderLogs;
    </script>
