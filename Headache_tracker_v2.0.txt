<script>
    const db = new Dexie("HeadacheDB");
    db.version(1).stores({ logs: '++id, date, type, sev, med' });

    let currentEntry = { id: null, type: '', sev: 0, med: [], date: null };

    // SET SINGLE VALUE (Type, Severity)
    function setSingle(field, val, btn) {
        currentEntry[field] = val;
        btn.parentElement.querySelectorAll('button').forEach(b => b.classList.remove('active'));
        btn.classList.add('active');
    }

    // TOGGLE MULTIPLE VALUES (Medication)
    function toggleMulti(field, val, btn) {
        if (val === 'None') {
            currentEntry[field] = ['None'];
            btn.parentElement.querySelectorAll('button').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
        } else {
            // Remove 'None' if other meds are selected
            currentEntry[field] = currentEntry[field].filter(m => m !== 'None');
            const noneBtn = Array.from(btn.parentElement.querySelectorAll('button')).find(b => b.innerText === 'None');
            noneBtn.classList.remove('active');

            if (currentEntry[field].includes(val)) {
                currentEntry[field] = currentEntry[field].filter(m => m !== val);
                btn.classList.remove('active');
            } else {
                currentEntry[field].push(val);
                btn.classList.add('active');
            }
        }
    }

    async function saveToDB() {
        if (!currentEntry.type || !currentEntry.sev) return alert("Select Type & Severity");
        
        const data = {
            type: currentEntry.type,
            sev: currentEntry.sev,
            med: currentEntry.med,
            date: currentEntry.id ? currentEntry.date : new Date() 
        };

        if (currentEntry.id) {
            await db.logs.update(currentEntry.id, data);
            alert("Entry Updated!");
        } else {
            await db.logs.add(data);
            alert("Logged!");
        }
        resetForm();
    }

    function resetForm() {
        currentEntry = { id: null, type: '', sev: 0, med: [], date: null };
        document.querySelectorAll('button').forEach(b => b.classList.remove('active'));
        document.getElementById('saveBtn').innerText = "Log Headache";
        document.getElementById('formTitle').innerText = "New Entry";
        document.getElementById('cancelBtn').style.display = "none";
        showPage('logPage', document.querySelector('.tab-item'));
    }

    async function deleteEntry(id) {
        if(confirm("Delete this entry?")) {
            await db.logs.delete(id);
            renderHistory();
        }
    }

    async function editEntry(id) {
        const item = await db.logs.get(id);
        currentEntry = { ...item };
        
        // Pre-fill UI
        showPage('logPage', document.querySelector('.tab-item'));
        document.getElementById('formTitle').innerText = "Edit Entry";
        document.getElementById('saveBtn').innerText = "Update Entry";
        document.getElementById('cancelBtn').style.display = "block";

        // Activate Buttons
        activateBtn('typeGrid', item.type);
        activateBtn('sevGrid', item.sev);
        item.med.forEach(m => activateBtn('medGrid', m));
    }

    function activateBtn(gridId, val) {
        const btns = document.querySelectorAll(`#${gridId} button`);
        btns.forEach(b => { if(b.innerText == val) b.classList.add('active'); });
    }

    async function renderHistory() {
        const logs = await db.logs.orderBy('date').reverse().toArray();
        const list = document.getElementById('historyList');
        list.innerHTML = logs.map(l => `
            <div class="card history-card">
                <div class="history-info">
                    <strong>${l.date.toLocaleDateString()}</strong><br>
                    ${l.type} | Sev: ${l.sev}<br>
                    <small>${l.med.join(', ') || 'No Meds'}</small>
                </div>
                <div class="history-actions">
                    <button class="btn-small btn-edit" onclick="editEntry(${l.id})">Edit</button>
                    <button class="btn-small btn-delete" onclick="deleteEntry(${l.id})">Del</button>
                </div>
            </div>
        `).join('');
    }

    function showPage(pageId, tab) {
        document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
        document.querySelectorAll('.tab-item').forEach(t => t.classList.remove('active'));
        document.getElementById(pageId).classList.add('active');
        tab.classList.add('active');
        if(pageId === 'historyPage') renderHistory();
        if(pageId === 'reportPage') renderReport();
    }

    // REPORTING LOGIC
    let myChart = null;
    async function renderReport() {
        const logs = await db.logs.orderBy('date').reverse().limit(30).toArray();
        const ctx = document.getElementById('headacheChart').getContext('2d');
        if (myChart) myChart.destroy();
        myChart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: logs.map(l => l.date.toLocaleDateString()).reverse(),
                datasets: [{ label: 'Severity', data: logs.map(l => l.sev).reverse(), borderColor: '#4a90e2', fill: false }]
            },
            options: { scales: { y: { min: 0, max: 5 } } }
        });
    }

    async function exportPDF() {
        const { jsPDF } = window.jspdf;
        const canvas = await html2canvas(document.getElementById('pdfContent'));
        const doc = new jsPDF();
        doc.text("Headache Report", 20, 20);
        doc.addImage(canvas.toDataURL('image/png'), 'PNG', 10, 30, 180, 100);
        doc.save("Report.pdf");
    }
</script>
