<script>
    // 1. DATABASE SETUP
    const db = new Dexie("HeadacheTrackerDB");
    db.version(1).stores({ logs: '++id, date, type, sev, med' });

    let entry = { type: '', sev: 0, med: [] };

    // 2. LOGGING LOGIC
    function setVal(f, v, el) {
        entry[f] = v;
        el.parentElement.querySelectorAll('button').forEach(b => b.classList.remove('active'));
        el.classList.add('active');
    }

    function toggleMed(v, el) {
        if (v === 'None') {
            entry.med = ['None'];
            el.parentElement.querySelectorAll('button').forEach(b => b.classList.remove('active'));
            el.classList.add('active');
        } else {
            entry.med = entry.med.filter(m => m !== 'None');
            const noneBtn = Array.from(el.parentElement.querySelectorAll('button')).find(b => b.innerText === 'None');
            noneBtn.classList.remove('active');
            
            if (entry.med.includes(v)) {
                entry.med = entry.med.filter(m => m !== v);
                el.classList.remove('active');
            } else {
                entry.med.push(v);
                el.classList.add('active');
            }
        }
    }

    async function saveLog() {
        if (!entry.type || !entry.sev) return alert("Please select Type and Severity");
        await db.logs.add({ ...entry, date: new Date() });
        alert("Saved Successfully");
        location.reload(); // Force clear and refresh for UI consistency
    }

    // 3. NAVIGATION & HISTORY FETCHING
    function nav(id, el) {
        document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
        document.querySelectorAll('.tab-item').forEach(t => t.classList.remove('active'));
        document.getElementById(id).classList.add('active');
        el.classList.add('active');
        
        if (id === 'historyPage') renderHistory();
        if (id === 'reportPage') renderCharts();
    }

    async function renderHistory() {
        const logs = await db.logs.orderBy('date').reverse().toArray();
        const container = document.getElementById('historyList');
        if (logs.length === 0) return;
        
        container.innerHTML = logs.map(l => `
            <div class="history-item">
                <div>
                    <strong>${new Date(l.date).toLocaleDateString()}</strong> - ${l.type}<br>
                    <small>Meds: ${l.med.join(', ')}</small>
                </div>
                <div style="font-weight:bold; color:var(--accent)">Sev: ${l.sev}</div>
                <button onclick="deleteEntry(${l.id})" style="padding:5px; background:none; color:red">✕</button>
            </div>
        `).join('');
    }

    async function deleteEntry(id) {
        if (confirm("Delete this entry?")) {
            await db.logs.delete(id);
            renderHistory();
        }
    }

    // 4. CHARTING & PDF
    let activeCharts = [];
    async function renderCharts(canvasId = 'uiChart') {
        const logs = await db.logs.orderBy('date').limit(30).toArray();
        const dates = logs.map(l => new Date(l.date).toLocaleDateString());
        
        const migraines = logs.map(l => l.type === 'Migraine' ? l.sev : null);
        const tensions = logs.map(l => l.type === 'Tension' ? l.sev : null);

        const ctx = document.getElementById(canvasId).getContext('2d');
        
        // Clear previous chart if exists
        activeCharts.forEach(c => { if(c.canvas.id === canvasId) c.destroy(); });

        const newChart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: dates,
                datasets: [
                    { label: 'Migraine', data: migraines, borderColor: '#0984e3', spanGaps: true, tension: 0.3 },
                    { label: 'Tension', data: tensions, borderColor: '#e67e22', spanGaps: true, tension: 0.3 }
                ]
            },
            options: { scales: { y: { min: 0, max: 5 } } }
        });
        activeCharts.push(newChart);
    }

    async function generatePDF() {
        // Prepare hidden PDF area
        const logs = await db.logs.orderBy('date').reverse().toArray();
        const tableBody = document.getElementById('pdfTableBody');
        tableBody.innerHTML = logs.map(l => `
            <tr>
                <td>${new Date(l.date).toLocaleDateString()}</td>
                <td>${l.type}</td>
                <td>${l.sev}</td>
                <td>${l.med.join(', ')}</td>
            </tr>
        `).join('');

        await renderCharts('pdfChart');

        // Capture and Download
        const area = document.getElementById('pdf-report-area');
        const canvas = await html2canvas(area, { scale: 2 });
        const imgData = canvas.toDataURL('image/png');
        
        const { jsPDF } = window.jspdf;
        const doc = new jsPDF('p', 'mm', 'a4');
        const imgProps = doc.getImageProperties(imgData);
        const pdfWidth = doc.internal.pageSize.getWidth();
        const pdfHeight = (imgProps.height * pdfWidth) / imgProps.width;
        
        doc.addImage(imgData, 'PNG', 0, 0, pdfWidth, pdfHeight);
        doc.save(`Headache_Report_${new Date().toLocaleDateString()}.pdf`);
    }
